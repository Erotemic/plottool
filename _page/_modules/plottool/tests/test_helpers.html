[docs]def dummy_bbox(img, shiftxy=(0.0, 0.0), scale=.25):
    """Default to rectangle that has a quarter-width/height border."""
    (gh, gw) = img.shape[0:2]
    half_w = gw / 2
    half_h = gh / 2
    w = (gw * scale)
    h = (gh * scale)
    x = half_w - (w / 2) + shiftxy[0] * gw
    y = half_h - (h / 2) + shiftxy[1] * gh
    bbox = tuple(map(int, map(round, [x, y, w, h])))
    return bbox
